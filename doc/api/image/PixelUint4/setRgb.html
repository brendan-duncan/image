@override
void setRgb(num r, num g, num b) {
  final nc = image.numChannels;
  if (nc > 0) {
    _setChannel(0, r);
    if (nc > 1) {
      _setChannel(1, g);
      if (nc > 2) {
        _setChannel(2, b);
      }
    }
  }
}
